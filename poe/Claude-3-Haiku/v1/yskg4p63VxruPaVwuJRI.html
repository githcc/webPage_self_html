    以下是一个使用 HTML、JavaScript 和 jsmediatags 库来实现读取和修改 MP3 文件标签信息的示例代码:

    html
    <!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>MP3 Tag Editor</title>
  <script src="https://cdn.jsdelivr.net/npm/jsmediatags/dist/jsmediatags.min.js"></script>
  <script src="https://unpkg.com/mp3tag.js/dist/mp3tag.min.js"></script>
</head>
<body>
  <h1>MP3 Tag Editor</h1>
  <input type="file" id="mp3-file" accept=".mp3">
  <button onclick="readTags()">Read Tags</button>
  <br>
  <label for="title">Title:</label>
  <input type="text" id="title" disabled>
  <br>
  <label for="artist">Artist:</label>
  <input type="text" id="artist" disabled>
  <br>
  <label for="album">Album:</label>
  <input type="text" id="album" disabled>
  <br>
  <label for="year">Year:</label>
  <input type="text" id="year" disabled>
  <br>
  <button onclick="updateTags()">Update Tags</button>
  <button onclick="saveAsNewFile()">Save as New File</button>

  <script>
    let originalFile;
    let originalTags;

    function readTags() {
      const fileInput = document.getElementById('mp3-file');
      const file = fileInput.files[0];
      originalFile = file;

      jsmediatags.read(file, {
        onSuccess: function(tag) {
          originalTags = tag.tags;
          document.getElementById('title').value = originalTags.title;
          document.getElementById('artist').value = originalTags.artist;
          document.getElementById('album').value = originalTags.album;
          document.getElementById('year').value = originalTags.year;
        },
        onError: function(error) {
          console.log('Something went wrong', error.type, error.info);
        }
      });
    }

    function updateTags() {
      const newTitle = document.getElementById('title').value;
      const newArtist = document.getElementById('artist').value;
      const newAlbum = document.getElementById('album').value;
      const newYear = document.getElementById('year').value;

      originalTags.title = newTitle;
      originalTags.artist = newArtist;
      originalTags.album = newAlbum;
      originalTags.year = newYear;
    }

    function saveAsNewFile() {
      const newTags = {
        title: document.getElementById('title').value,
        artist: document.getElementById('artist').value,
        album: document.getElementById('album').value,
        year: document.getElementById('year').value
      };

      const mp3tag = new MP3Tag(originalFile);
      mp3tag.setTags(newTags);
      mp3tag.save('new_mp3_file.mp3');
    }
  </script>
</body>
</html>

    这个示例包含以下功能:

    1. 选择一个 MP3 文件并点击 "Read Tags" 按钮，会读取文件的标签信息并显示在相应的输入框中。
    2. 用户可以编辑标签信息,然后点击 "Update Tags" 按钮来修改原始标签信息。
    3. 用户可以点击 "Save as New File" 按钮,将修改后的标签信息保存到一个新的 MP3 文件中(new_mp3_file.mp3)。

    这个示例使用了 jsmediatags 库来读取 MP3 文件的标签信息,并使用 mp3tag.js 库来保存修改后的标签信息到一个新的 MP3 文件中。你可以根据需要进一步扩展和优化这个代码。
