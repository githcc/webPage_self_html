    <template>
  <div id="app">
    <div class="container">
      <div class="sidebar">
        <h2>Conversations</h2>
        <ul id="conversation-list">
          <li
            v-for="(conversation, index) in conversations"
            :key="index"
            :class="{ active: index === currentConversation }"
            @click="displayConversation(index)"
          >
            {{ conversation.name }}
          </li>
        </ul>
      </div>
      <div class="chat-container">
        <div class="message-container" id="message-container">
          <div
            v-for="(message, index) in conversations[currentConversation].messages"
            :key="index"
            class="message"
          >
            [{{ conversations[currentConversation].name }}] {{ message.sender }}: {{ message.text }}
          </div>
        </div>
        <div class="input-container">
          <textarea
            id="message-input"
            v-model="currentMessage"
            placeholder="Type your message..."
          ></textarea>
          <button id="send-button" @click="sendMessage">Send</button>
        </div>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  name: 'App',
  data() {
    return {
      currentConversation: 0,
      currentMessage: '',
      conversations: [
        {
          name: 'Emoji',
          prompt:
            ' emoji  emoji \n\n\n',
          messages: []
        },
        {
          name: 'Conversation 2',
          prompt: 'custom prompt',
          messages: []
        }
      ]
    };
  },
  methods: {
    displayConversation(index) {
      this.currentConversation = index;
    },
    sendMessage() {
      if (this.currentMessage.trim()) {
        this.addMessage('You', this.currentMessage);
        this.conversations[this.currentConversation].messages.push({
          sender: 'You',
          text: this.currentMessage
        });
        this.currentMessage = '';
      }
    },
    addMessage(sender, text) {
      const messageElement = document.createElement('div');
      messageElement.classList.add('message');
      messageElement.textContent = `[${this.conversations[this.currentConversation].name}] ${sender}: ${text}`;
      this.$el.querySelector('#message-container').appendChild(messageElement);
    }
  }
};
</script>
